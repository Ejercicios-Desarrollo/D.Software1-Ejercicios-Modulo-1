#include "pch.h"
#include <iostream>
#include <sstream>
using namespace std;

struct persona {
	int edad;
	string nombre;
};


int main()
{
	//dada n personas ingresar edad y ordenar de menor a mayor;

	//int mayor;
	int n = 3;
	
	/*cout << "Ingresa cantidad de personas" << endl;
	cin >> n;*/
	struct persona p[3];

	for (int i = 0; i < n; i++)
	{
		cout << "Ingresa nombre" << endl;
		cin >> p[i].nombre;
		cout << "Ingresa edad" << endl;
		cin >> p[i].edad;
	}

	for (int i = 1; i <= n - 1; i++) {
		for (int j = 1; j <= n - i; j++) {
			if (p[j - 1].edad < p[j].edad) {
				persona aux = p[j];
				p[j] = p[j - 1];
				p[j - 1] = aux;
			}
		}
	}


	for (int i = 0; i < n; i++)
	{
		cout << p[i].nombre <<": " << p[i].edad << endl;
	}


	

}
